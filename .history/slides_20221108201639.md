```haskell
class Eq α where
  eq :: α -> α -> Bool

instance Eq Nat where
  eq Zero    Zero    = True
  eq (Suc x) (Suc y) = eq x y
  eq _       _       = False



instance Eq α => Eq [α] where
  eq []       []       = True
  eq (x : xs) (y : ys) = eq x y && 
                         eq xs ys
  eq _        _        = False



isEq :: Bool
isEq = eq [Zero] [Zero]
```

```rust
trait Eq
  fn eq(&self, rhs: &Self) -> Bool

impl Eq for Nat
  fn eq(&self, rhs: &Self) -> Bool
    match (self, rhs)
      (Zero, Zero)     => True,
      (Suc(x), Suc(y)) => x.eq(y),
      (_, _)           => False

impl<A: Eq> Eq for [A]
  fn eq(&self, rhs: &Self) -> Bool 
    match (self, rhs) 
      ([], []) => True,
      ([x, xs@..], [y, ys@..]) 
        => x.eq(y) && xs.eq(ys),
      (_ , _)  => False

fn is_eq() -> Bool 
  [Zero].eq(&[Zero])
```

```haskell
inst eq :: Nat -> Nat -> Bool
  eq Zero    Zero    = True
  eq (Suc x) (Suc y) = eq x y
  eq _       _       = False

inst eq :: (eq :: α -> α -> Bool) => [α] -> [α] -> Bool
  eq []     []       = True
  eq (x:xs) (y:ys)   = eq x y && eq xs ys
  eq _      _        = False

fun isEq :: Bool
  isEq = [Zero] == [Zero]
```
